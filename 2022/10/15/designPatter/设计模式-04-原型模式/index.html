设计模式:04--原型模式

2022-10-15 模式设计 模式设计 0 Comments Word Count: 615(words) Read Count: 3(minutes)

一、概念及UML图

1、概念

用原型实例指定创建对象的种类，并且通过拷贝这些原型来创建新的对象

2、应用场景

用new创建一个对象需要非常繁琐的数据准备或者权限，可以考虑使用原型模式

#include <iostream>
#include <functional>
#include <memory>
#include <string>
#include <vector>
#include <unordered_map>
#include <cassert>

using namespace std;

class IProduct {
public:
    virtual ~IProduct() {}
    virtual IProduct *Clone() = 0;
    virtual void Use(const std::string &str) = 0;
};

class MessageBox : public IProduct {
public:
    MessageBox(const std::string &str) : m_name(str) {}
    IProduct *Clone() override {
        return new MessageBox(m_name);
    }
    void Use(const std::string &str) {
        std::cout << "Message box " << m_name << " " << str << std::endl;
    }

private:
    std::string m_name;
};

class ProductMgr {
public:
    void Register(const std::string &name, IProduct *pro) {
        m_map.insert({ name, pro });
    }
    IProduct *Create(const std::string &name) {
        auto it = m_map.find(name);
        if (it != m_map.end())
            return it->second->Clone();
        else
            return nullptr;
    }

private:
    std::unordered_map<std::string, IProduct *> m_map;
};

int main()
{
    ProductMgr mgr;
    IProduct *p1 = new MessageBox("Hello");
    mgr.Register("MB", p1);
    p1->Use("World");
    auto p2 = mgr.Create("MB");
    p2->Use("World");
    cout << p1 << endl;
    cout << p2 << endl;

    auto pp = mgr.Create("Not");
    assert(pp == nullptr);

    delete p1;
    delete p2;

    return 0;
}
